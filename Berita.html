linkberita = 'https://news.kompas.com/search/all/'
listt = []
n = 5

for page in range(1, n+1):
  req = requests.get(linkberita)
  soup = BeautifulSoup(req.text, 'html.parser')
  items = soup.findAll('div', 'article__list clearfix')
  for it in items:
    link = it.find('a','article__link')['href']

    req2 = requests.get(link)
    soup2 = BeautifulSoup(req2.text, 'html.parser')
    judul = soup2.find('h1', 'read__title').text

    print(judul)
    print(isi)
    listt.append([judul, isi])
